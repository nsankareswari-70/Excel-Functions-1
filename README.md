# Excel-Functions-1

# And Function
Check if the Pokemon type is fire and has speed greater than 70

=AND(B2:B10="Fire",C2:C10>70)

# If and And function - combination

=IF(AND(B2:B10="Fire",C2:C10>70),"Yes","No")
